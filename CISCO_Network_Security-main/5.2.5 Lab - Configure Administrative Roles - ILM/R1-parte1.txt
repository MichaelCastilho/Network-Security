!Autor: Michael Castilho
!Curso Network Security
!Utilize o Cisco Packet Tracer 8.2.1
!Configuracoes do R1


enable
configure terminal

!Atribuindo nome ao dispositivo
	hostname R1

!Configurando Interfaces
	interface g0/0/0
	 ip address 10.1.1.1 255.255.255.252
	 no shutdown
	 exit

	interface g0/0/1
	 ip address 192.168.1.1 255.255.255.0
	 no shutdown
	 exit

!Desative a pesquisa de DNS no R1 / Shift F6
	no ip domain-lookup

!Roteamento OSPF no R1

	router ospf 1
	 network 10.1.1.0 0.0.0.3 area 0
	 network 192.168.1.0 0.0.0.255 area 0
	 passive-interface g0/0/1
	 exit

!Permitir AAA no R1 - Lab pede mas ser√° visto mais pra frente
	aaa new-model

!Senha modo EXEC Privilegiado
	enable secret cisco12345
